``insertion sort``

{insrt list: [n_n] -> list |
	i: [n < # list] = 0
	j: [n < # list]
	swap = ( .
		? list_j < list_ (j - 1)
		( 
			(list_j list_ (j - 1) ) 
				= 
			(list_ (j - 1) list_j)

			? j > 0 (j = j - 1 (sort) ) ()
		) (/)
	)
	( loop = ( .
		? i < # loop 
		( 
			i = i + 1
			j = i
			(swap)
			(loop)
		) (/)
	)
}
